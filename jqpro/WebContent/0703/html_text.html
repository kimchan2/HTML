3. html/text

$("p").html() : 첫번째 p태그를 대상으로 내용을 가져온다
모든 p태그를 대상으로 실행할때는 반복문을 사용
반복문안에서 this로 각p태그를 접근한다
// 반복문
$.each($(".box3 p"), function() {
res += $(this).html();
})
$.each($(".box3 p"), function(i) {
res += i + "번째 : " + $(this).html();
})
$("#result3").html(값) : #result3에 값 출력
